POST /login
	Request 
		Body:
			{
				"login": string,
				"password": string
			}
		
	Response 
		Headers:
			Set-Cookie: Authorization={token}            // Authorization token
		Body:
			{
				"isSuccess": bool,
				"error" : {
					"code": number,
					"message": string
				}
			}
		
POST /registration
	Request:
		Body:
			{
				"login": string,
				"password": string,
				"email": string
			}
		
	Response: 
		Body:
			{
				"isSuccess": bool,
				"error" : {
					"code": number,
					"message": string
				}
			}

GET /userData
	Request 
		Headers:
			Cookies: Authorization=token                 // Requires token
		
	Response 
		Body:
			{
				"isSuccess": bool,
				"error" : {
					"code": number,
					"message": string
				},
				"userData": {
					"id": string,
					"login": string,
					"email": number
				}
			}
			
GET /groups
	Request 
		Headers:
			Cookies: Authorization=token                 // Requires token
		
	Response 
		Body:
			{
				"isSuccess": bool,
				"error" : {
					"code": number,
					"message": string
				},
				"groups": [{
					"id": string,
					"name": string,
					"totalWords": number,
					"learnedWords": number,
					"previewImageUrl": string
				}]
			}
			
POST /words
	Request 
			Cookies: Authorization=token
		Body:
			{
				"mode": string ("group"/"learned"/"all"),        // 3 possible values. If {"mode": "group"}, need to set "groupId"
				"groupId": string
			}
		
	Response 
		Body:
			{
				"isSuccess": bool,
				"error" : {
					"code": number,
					"message": string
				},
				"words": [{
					"id": string,
					"nativeTranslation": string,        // Russian word
					"foreignTranslations": string[],    // English words
					"imageUrl": string
				}]
			}
			
POST /commit
	Request 
			Cookies: Authorization=token
		Body:
			{
				"words": [{
					"wordId": string,
					"learnAttempts": string,
					"isStudied": boolean
				}]
			}
		
	Response 
		Body:
			{
				"isSuccess": bool,
				"error" : {
					"code": number,
					"message": string
				}
			}
