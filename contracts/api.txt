- - - - - - -
All methods which return JSON have next fields:

"isSuccess": bool,
"error" : {
	"code": number,
	"message": string
}

- - - - - - -

POST /api/login
	Request 
		Body:
			{
				"login": string,
				"password": string
			}
		
	Response 
		Headers:
			Set-Cookie: Authorization={token}            // Authorization token
		Body:
			{
				"isSuccess": bool,
				"error" : {
					"code": number,
					"message": string
				}
			}

POST /api/getUserData
	Request 
		Headers:
			Cookies: Authorization=token                 // Requires token
		
	Response 
		Body:
			{
				"isSuccess": bool,
				"error" : {
					"code": number,
					"message": string
				},
				"userData": {
					"id": string,
					"login": string,
					"email": number,
					"nativeLangCode": string,
					"foreignLangCode": string
				}
			}
			
POST /api/getGroups
	Request 
		Headers:
			Cookies: Authorization=token                 // Requires token
		
	Response 
		Body:
			{
				"isSuccess": bool,
				"error" : {
					"code": number,
					"message": string
				},
				"groups": [{
					"id": string,
					"name": string,
					"totalWords": number,
					"learnedWords": number,
					"previewImageUrl": string
				}]
			}
			
POST /api/getWords
	Request 
		Headers:
			Cookies: Authorization=token
		Body:
			{
				"mode": string ("group"/"learned"/"all"),        // 3 possible values. If {"mode": "group"}, need to set "groupId"
				"groupId": string
			}
		
	Response 
		Body:
			{
				"isSuccess": bool,
				"error" : {
					"code": number,
					"message": string
				},
				"nativeLangCode": string,                // RU
				"foreignLangCode": string,               // EN
				"words": [{
					"id": string,
					"name": string,  
					"nativeTranslation": string,     // Russian word
					"foreignTranslations": [{		 // English words
						"translation": string,
						"priority": number
					}],
					"imageUrl": string
				}]
			}
			
POST /api/commitWords
	Request 
		Headers:
			Cookies: Authorization=token
		Body:
			{
				"words": [{
					"wordId": string,
					"learnAttempts": string,
					"isStudied": boolean
				}]
			}
		
	Response 
		Body:
			{
				"isSuccess": bool,
				"error" : {
					"code": number,
					"message": string
				}
			}

// Admin methods
			
POST /api/admin/addWords
	Request 
		Headers:
			Cookies: Authorization=token
		Body:
			{
				"words": [{
					"id": string? [GUID],
					"name": string,
					"translations": [{
						"langCode": string ("RU"/"EN"/"DE"),
						"translation": string,
						"priority": number?
					}]
				}]
			}
		
	Response 
		Body:
			{
				"isSuccess": bool,
				"error" : {
					"code": number,
					"message": string
				},
				"addedWords": {
					"id": string,
					"name": string
				}
			}
